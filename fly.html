function doPost(e) {
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
  var data = JSON.parse(e.postData.contents);
  sheet.appendRow([
    new Date(),
    data.className,
    data.seatNo,
    data.studentName,
    data.levelName,
    data.stepCount,
    data.commands,
    data.status
  ]);
  return ContentService.createTextOutput(JSON.stringify({"status": "success"}))
    .setMimeType(ContentService.MimeType.JSON);
}

function doGet(e) {
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
  sheet.appendRow([
    new Date(),
    e.parameter.className,
    e.parameter.seatNo,
    e.parameter.studentName,
    e.parameter.levelName,
    e.parameter.stepCount,
    e.parameter.commands,
    e.parameter.status
  ]);
  return ContentService.createTextOutput(e.parameter.callback + "(" + JSON.stringify({"status": "success"}) + ")")
    .setMimeType(ContentService.MimeType.JAVASCRIPT);
}
